"use strict"

/**
 * counter
 */

function counter() {
  let seconds = 0
  setInterval(() => {
    seconds += 1
    document.getElementById(
      "app"
    ).innerHTML = `<p>You have been here for ${seconds} seconds.</p>`
  }, 1000)
}
